//  BUTTONS

	Buttons

	//-------------------
	//  SHIP
	//-------------------

		*btnShip
			name:Your starship
			desc:Click to perform manual engine burst
			icon:https://gamingdark.github.io/nmc/idm/img/ship.png
			on click:
				yield 1 distance
				if (chance(10%)) yield 3 resWork
				if (chance(1%)) yield 1 resScience
				if (chance(1%)) yield 1 resTech
			end
			no tooltip
			no text
			class:bigButton
